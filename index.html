☁️ 여러 기기 동기화
같은 URL을 다른 기기(폰/PC)의 Folio에도 똑같이 설정해두면, 구글시트에 데이터를 올리고 받아오는 방식으로 동기화할 수 있어요. 자동 동기화는 아니고, 작업 끝낸 기기에서 업로드, 새로 시작하는 기기에서 다운로드를 직접 눌러주는 수동 방식이에요.
  1. script.google.com 접속 → 새 프로젝트
  2. 아래 코드를 전체 복사해서 기본 코드를 모두 지우고 붙여넣기
  3. 우측 상단 배포 → 새 배포 → 유형: 웹 앱
  4. 실행 계정: 나  /  액세스 권한: 모든 사용자 선택 후 배포
  5. 발급된 웹 앱 URL을 복사해서 위 입력칸에 붙여넣고 저장
function doGet(e){
  if(e.parameter.action === 'loadDB') return handleLoadDB();
  var tickers = (e.parameter.tickers||'').split(',').map(function(s){return s.trim();}).filter(Boolean);
  var out = {};
  var fxRate = null; // USD→KRW 환율, 해외종목이 있을 때만 1회 조회해서 재사용
  tickers.forEach(function(t){
    try{
      if(/^[0-9A-Za-z]{6}$/.test(t)){
        out[t] = getKrPrice(t);
      } else {
        if(fxRate === null) fxRate = getUsdKrwRate();
        out[t] = getUsPrice(t, fxRate);
      }
    }catch(err){ out[t] = null; }
  });
  return ContentService.createTextOutput(JSON.stringify(out))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e){
  if(e.parameter.action === 'saveDB') return handleSaveDB(e.postData.contents);
  return ContentService.createTextOutput(JSON.stringify({error:'unknown action'}))
    .setMimeType(ContentService.MimeType.JSON);
}

function getKrPrice(code){
  var url = 'https://finance.naver.com/item/main.naver?code=' + code;
  var html = UrlFetchApp.fetch(url, {muteHttpExceptions:true}).getContentText('UTF-8');
  var m = html.match(/class="no_today"[\s\S]*?class="blind">([\d,]+)<\/span>/);
  return m ? Number(m[1].replace(/,/g,'')) : null;
}

function getUsPrice(ticker, fxRate){
  var url = 'https://query1.finance.yahoo.com/v8/finance/chart/' + encodeURIComponent(ticker);
  var json = JSON.parse(UrlFetchApp.fetch(url, {muteHttpExceptions:true}).getContentText());
  var r = json.chart && json.chart.result && json.chart.result[0];
  if(!r) return null;
  var usd = r.meta.regularMarketPrice;
  return fxRate ? Math.round(usd * fxRate) : usd; // 원화로 환산
}

function getUsdKrwRate(){
  var url = 'https://query1.finance.yahoo.com/v8/finance/chart/KRW=X';
  var json = JSON.parse(UrlFetchApp.fetch(url, {muteHttpExceptions:true}).getContentText());
  var r = json.chart && json.chart.result && json.chart.result[0];
  return r ? r.meta.regularMarketPrice : null;
}

function getDataSheet(){
  var props = PropertiesService.getScriptProperties();
  var ssId  = props.getProperty('FOLIO_SHEET_ID');
  var ss;
  if(ssId){
    ss = SpreadsheetApp.openById(ssId);
  } else {
    ss = SpreadsheetApp.create('Folio_Data');
    props.setProperty('FOLIO_SHEET_ID', ss.getId());
  }
  return ss.getSheetByName('FolioDB') || ss.insertSheet('FolioDB');
}
